Table players {
    player_id int [pk, increment]
    uniform_number int
    player_name varchar(45)
    player_country varchar(45)
    player_active varchar(45)
    player_il varchar(45)
    player_age int
    player_hitting varchar(45)
    player_throwing varchar(45)
    player_height varchar(45)
    player_weight int
    player_dob varchar(45)
    player_firstyear int
    player_nameadditional varchar(45)
    player_position varchar(45)
}

Table team_stats {
    team_id int [pk, increment]
    team_name varchar(100)
    team_abbreviation varchar(3)
    season_year int
    games_played int
    wins int
    losses int
    runs int
    home_runs int
    batting_average decimal(5,3)
    on_base_percentage decimal(5,3)
    slugging_percentage decimal(5,3)
    era decimal(4,2)
    whip decimal(5,3)
    strikeouts int
    save_opportunities int
    stolen_bases int
    fielding_percentage decimal(5,3)
    last_updated timestamp
}

Table polls {
    poll_id int [pk, increment]
    question varchar(255)
}

Table poll_options {
    option_id int [pk, increment]
    poll_id int [ref: > polls.poll_id]
    poll_option varchar(255)
    votes int
}
